function isInViewport(el) {
    const rect = el.getBoundingClientRect();
    return (
        rect.left >= -1 * (window.innerWidth/3) && 
        rect.left <= (window.innerWidth / 3) &&
        rect.right >= window.innerWidth * 2 / 3
    );
}

document.querySelectorAll('a[href^="#"]').forEach(anchor => {

        });
